# ====================================================================
#  CORE SPIKING NEURAL NETWORK ENGINE
# ====================================================================

@dataclass
class Spike:
    """A single neural spike/event"""
    neuron_id: str
    timestamp: float
    strength: float = 1.0
    source: str = "external"

@dataclass
class Neuron:
    """A single neuron in the spiking network"""
    id: str
    threshold: float = 100.0
    leak_rate: float = 2.0
    membrane_potential: float = 0.0
    last_spike_time: float = 0.0
    refractory_period: float = 50.0  # milliseconds
    connections: List[str] = field(default_factory=list)
    weights: Dict[str, float] = field(default_factory=dict)
    spike_count: int = 0
    concept: str = ""
    importance: float = 1.0

@dataclass
class KnowledgeChunk:
    """A piece of knowledge from textbooks"""
    id: str
    content: str
    concepts: List[str]
    source: str
    importance: float
    embedding: Optional[List[float]] = None

class SpikingKnowledgeEngine:
    """
    Core spiking neural network for educational AI
    Uses spike-timing-dependent plasticity (STDP) for learning
    """
    
    def __init__(self):
        self.neurons: Dict[str, Neuron] = {}
        self.knowledge_base: Dict[str, KnowledgeChunk] = {}
        self.concept_index: Dict[str, List[str]] = defaultdict(list)
        self.spike_history: List[Spike] = []
        self.learning_rate: float = 0.01
        self.simulation_time: float = 0.0
        self.student_profiles: Dict[str, Dict] = {}
        self.current_student: Optional[str] = None
        
        # Mesh networking
        self.mesh_peers: Dict[str, Any] = {}
        self.shared_knowledge: Dict[str, Any] = {}
        
        # Performance metrics
        self.metrics = {
            "spikes_processed": 0,
            "answers_generated": 0,
            "energy_used": 0.0,  # simulated energy units
            "questions_answered": 0
        }
    
    # ============================================================
    #  KNOWLEDGE LOADING & EMBEDDING
    # ============================================================
    
    def load_textbook(self, textbook_data: Dict, source_name: str):
        """
        Load a textbook into the spiking network
        Each concept becomes a neuron that spikes when relevant
        """
        for chapter in textbook_data.get("chapters", []):
            for concept in chapter.get("concepts", []):
                # Create neuron for this concept
                neuron_id = f"{source_name}_{concept['name']}"
                
                # Importance determines threshold (higher = more important)
                importance = concept.get("importance", 0.5)
                threshold = 100 - (importance * 30)  # More important = easier to fire
                
                neuron = Neuron(
                    id=neuron_id,
                    threshold=max(50, threshold),
                    leak_rate=max(1, 5 - importance * 4),
                    concept=concept['name'],
                    importance=importance
                )
                
                # Store knowledge chunk
                knowledge = KnowledgeChunk(
                    id=neuron_id,
                    content=concept.get("explanation", ""),
                    concepts=[concept['name']] + concept.get("related", []),
                    source=f"{source_name} - {chapter.get('title', 'Unknown')}",
                    importance=importance
                )
                
                self.neurons[neuron_id] = neuron
                self.knowledge_base[neuron_id] = knowledge
                
                # Index concepts
                for concept_name in knowledge.concepts:
                    self.concept_index[concept_name.lower()].append(neuron_id)
        
        # Build connections between related concepts
        self._build_connections()
        print(f"✅ Loaded {len(self.neurons)} knowledge neurons from {source_name}")
    
    def load_knowledge_pack(self, pack_path: str):
        """
        Load a complete knowledge pack (JSON format)
        """
        with open(pack_path, 'r') as f:
            data = json.load(f)
        
        for source, textbook in data.items():
            self.load_textbook(textbook, source)
        
        print(f"📚 Knowledge pack loaded: {len(self.knowledge_base)} chunks")
    
    def _build_connections(self):
        """
        Build weighted connections between related neurons
        Using spike-timing-dependent plasticity principles
        """
        concept_pairs = defaultdict(list)
        
        # Group neurons by concepts
        for neuron_id, knowledge in self.knowledge_base.items():
            for concept in knowledge.concepts:
                concept_pairs[concept.lower()].append(neuron_id)
        
        # Create connections between neurons that share concepts
        for concept, neurons in concept_pairs.items():
            for i, neuron_a in enumerate(neurons):
                for neuron_b in neurons[i+1:]:
                    # Weight based on concept importance
                    weight = (self.neurons[neuron_a].importance + 
                             self.neurons[neuron_b].importance) * 0.5
                    
                    self.neurons[neuron_a].connections.append(neuron_b)
                    self.neurons[neuron_a].weights[neuron_b] = weight
                    self.neurons[neuron_b].connections.append(neuron_a)
                    self.neurons[neuron_b].weights[neuron_a] = weight
    
    # ============================================================
    #  TEXT TO SPIKE CONVERSION
    # ============================================================
    
    def text_to_spikes(self, text: str) -> List[Spike]:
        """
        Convert a question/query to a spike pattern
        Each keyword generates a spike burst
        """
        spikes = []
        words = re.findall(r'\b\w+\b', text.lower())
        
        for word in words:
            # Check if this word maps to a concept
            if word in self.concept_index:
                # Generate spikes for each matching neuron
                for neuron_id in self.concept_index[word][:3]:  # Limit to top matches
                    # Strength based on relevance
                    strength = 0.7 + random.random() * 0.3
                    spikes.append(Spike(
                        neuron_id=neuron_id,
                        timestamp=self.simulation_time,
                        strength=strength,
                        source="text_processing"
                    ))
            
            # Also check for partial matches
            for concept in self.concept_index:
                if word in concept and len(word) > 3:
                    for neuron_id in self.concept_index[concept][:2]:
                        spikes.append(Spike(
                            neuron_id=neuron_id,
                            timestamp=self.simulation_time,
                            strength=0.5,
                            source="fuzzy_match"
                        ))
        
        # Add random "noise" spikes (makes network more robust)
        if random.random() < 0.05:
            spike_neurons = list(self.neurons.keys())
            noise_neuron = random.choice(spike_neurons)
            spikes.append(Spike(
                neuron_id=noise_neuron,
                timestamp=self.simulation_time,
                strength=0.1,
                source="background_noise"
            ))
        
        return spikes
    
    def question_to_spikes(self, question: str) -> List[Spike]:
        """
        Specialized spike conversion for questions
        """
        spikes = self.text_to_spikes(question)
        
        # Add temporal encoding (spikes closer together = more urgent)
        now = self.simulation_time
        for i, spike in enumerate(spikes):
            # Slight delay variation based on question structure
            spike.timestamp = now + (i * 0.1) + random.random() * 0.05
        
        return spikes
    
    # ============================================================
    #  SPIKE PROCESSING ENGINE
    # ============================================================
    
    def process_spikes(self, spikes: List[Spike], student_id: Optional[str] = None):
        """
        Main processing loop - inject spikes and process through network
        """
        self.current_student = student_id
        start_time = time.time()
        
        # Clear previous spikes
        self.simulation_time += 1.0
        
        # Process each spike
        for spike in spikes:
            self._inject_spike(spike)
            
            # Update metrics
            self.metrics["spikes_processed"] += 1
            
            # Simulated energy usage (event-driven)
            self.metrics["energy_used"] += 0.001  # Very low power!
        
        # Let network "simmer" - process through connections
        for _ in range(5):  # 5 simulation cycles
            self._simulate_cycle()
        
        # Read output spikes
        fired_neurons = self._read_fired_neurons()
        
        # Process time
        elapsed = time.time() - start_time
        self.metrics["questions_answered"] += 1
        
        return self._generate_answer(fired_neurons)
    
    def _inject_spike(self, spike: Spike):
        """
        Inject a spike into the network
        """
        if spike.neuron_id not in self.neurons:
            return
        
        neuron = self.neurons[spike.neuron_id]
        
        # Check refractory period
        refractory_ms = (self.simulation_time - neuron.last_spike_time) * 1000
        if refractory_ms < neuron.refractory_period:
            return  # Neuron is in refractory period
        
        # Add to membrane potential (with weight)
        potential_increase = spike.strength * 20
        neuron.membrane_potential += potential_increase
        
        # Check if threshold reached
        if neuron.membrane_potential >= neuron.threshold:
            self._fire_neuron(spike.neuron_id, spike.strength)
            neuron.membrane_potential = 0  # Reset after firing
            neuron.last_spike_time = self.simulation_time
            neuron.spike_count += 1
            
            # Record spike in history
            self.spike_history.append(spike)
            
            # Propagate to connected neurons
            for connection in neuron.connections:
                weight = neuron.weights.get(connection, 0.5)
                if weight > 0.3:  # Only propagate significant weights
                    propagated_spike = Spike(
                        neuron_id=connection,
                        timestamp=self.simulation_time,
                        strength=spike.strength * weight * 0.3,
                        source=f"propagated_from_{spike.neuron_id}"
                    )
                    self._inject_spike(propagated_spike)
    
    def _fire_neuron(self, neuron_id: str, strength: float):
        """
        Neuron fires (spikes)
        """
        neuron = self.neurons[neuron_id]
        
        # STDP Learning: Update weights based on timing
        for connection in neuron.connections:
            connected_neuron = self.neurons[connection]
            if connected_neuron.last_spike_time > 0:
                # If connected neuron fired recently, strengthen connection
                time_diff = abs(connected_neuron.last_spike_time - self.simulation_time)
                if time_diff < 0.5:  # Within 500ms
                    # Hebbian learning: neurons that fire together, wire together
                    weight_increase = self.learning_rate * strength * 0.5
                    current_weight = neuron.weights.get(connection, 0.5)
                    new_weight = min(1.0, current_weight + weight_increase)
                    neuron.weights[connection] = new_weight
    
    def _simulate_cycle(self):
        """
        One simulation cycle - leak potentials and background processing
        """
        self.simulation_time += 0.01
        
        for neuron in self.neurons.values():
            # Leak membrane potential over time
            if neuron.membrane_potential > 0:
                neuron.membrane_potential -= neuron.leak_rate * 0.1
                neuron.membrane_potential = max(0, neuron.membrane_potential)
            
            # Random background spike (brain noise)
            if random.random() < 0.001:
                self._inject_spike(Spike(
                    neuron_id=neuron.id,
                    timestamp=self.simulation_time,
                    strength=0.05,
                    source="background"
                ))
    
    def _read_fired_neurons(self) -> List[str]:
        """
        Read which neurons fired in the last cycle
        """
        fired = []
        threshold_ms = 0.5  # Last 500ms
        
        for spike in reversed(self.spike_history):
            if self.simulation_time - spike.timestamp < threshold_ms:
                if spike.neuron_id not in fired:
                    fired.append(spike.neuron_id)
            else:
                break  # History is sorted by time
        
        return fired
    
    # ============================================================
    #  ANSWER GENERATION
    # ============================================================
    
    def _generate_answer(self, fired_neurons: List[str]) -> Dict:
        """
        Generate answer from fired neurons
        """
        if not fired_neurons:
            return {
                "answer": "I'm not sure. That doesn't seem to be in our curriculum.",
                "confidence": 0.0,
                "sources": [],
                "concepts": []
            }
        
        # Find knowledge chunks for fired neurons
        answers = []
        total_confidence = 0
        
        for neuron_id in fired_neurons:
            if neuron_id in self.knowledge_base:
                knowledge = self.knowledge_base[neuron_id]
                neuron = self.neurons[neuron_id]
                
                # Calculate confidence based on spike strength
                confidence = min(1.0, neuron.spike_count / 10)
                
                answers.append({
                    "content": knowledge.content,
                    "source": knowledge.source,
                    "concept": knowledge.concepts[0] if knowledge.concepts else "",
                    "confidence": confidence,
                    "neuron_id": neuron_id
                })
                
                total_confidence += confidence
        
        # Sort by confidence
        answers.sort(key=lambda x: x["confidence"], reverse=True)
        
        # Construct best answer
        best_answer = answers[0] if answers else None
        
        if best_answer and best_answer["confidence"] > 0.6:
            # Include related concepts
            related = []
            for neuron_id in fired_neurons:
                if neuron_id in self.knowledge_base:
                    concepts = self.knowledge_base[neuron_id].concepts
                    related.extend([c for c in concepts if c != best_answer["concept"]])
            
            return {
                "answer": best_answer["content"],
                "confidence": best_answer["confidence"],
                "sources": [best_answer["source"]],
                "concepts": [best_answer["concept"]] + related[:3],
                "all_answers": answers[:3]
            }
        else:
            # Low confidence - suggest related topics
            return {
                "answer": f"I found some information, but I'm not sure if it's right. Did you mean: {', '.join([a['concept'] for a in answers[:3]])}?",
                "confidence": total_confidence / len(answers) if answers else 0,
                "sources": [a["source"] for a in answers[:2]],
                "concepts": [a["concept"] for a in answers[:3]],
                "all_answers": answers[:2]
            }
    
    # ============================================================
    #  STUDENT ADAPTATION
    # ============================================================
    
    def adapt_to_student(self, student_id: str):
        """
        Load student profile and adapt network
        """
        self.current_student = student_id
        
        if student_id not in self.student_profiles:
            self.student_profiles[student_id] = {
                "started": datetime.now().isoformat(),
                "questions_asked": 0,
                "concepts_strengthened": {},
                "learning_style": "balanced",
                "difficulty_comfort": 0.5
            }
        
        profile = self.student_profiles[student_id]
        
        # Adjust network based on student profile
        for concept, strength in profile["concepts_strengthened"].items():
            # Find neurons for this concept
            if concept.lower() in self.concept_index:
                for neuron_id in self.concept_index[concept.lower()]:
                    # Lower threshold = easier to fire
                    self.neurons[neuron_id].threshold = max(50, 
                        100 - (strength * 20))
    
    def record_learning(self, concept: str, success: bool):
        """
        Record student learning and adjust network
        """
        if not self.current_student:
            return
        
        profile = self.student_profiles[self.current_student]
        
        if concept not in profile["concepts_strengthened"]:
            profile["concepts_strengthened"][concept] = 0
        
        if success:
            profile["concepts_strengthened"][concept] += 0.1
            profile["concepts_strengthened"][concept] = min(1.0, 
                profile["concepts_strengthened"][concept])
        else:
            # If struggling, don't penalize too much
            profile["concepts_strengthened"][concept] = max(0,
                profile["concepts_strengthened"][concept] - 0.05)
        
        profile["questions_asked"] += 1
        
        # Adapt network for this student
        self.adapt_to_student(self.current_student)
    
    # ============================================================
    #  MESH NETWORK SYNCHRONIZATION
    # ============================================================
    
    def export_knowledge(self) -> Dict:
        """
        Export learned knowledge for sharing across mesh network
        """
        export_data = {
            "timestamp": datetime.now().isoformat(),
            "neurons": {},
            "connections": {},
            "student_profiles": {}
        }
        
        # Export neuron weights and thresholds (the "learning")
        for neuron_id, neuron in self.neurons.items():
            export_data["neurons"][neuron_id] = {
                "threshold": neuron.threshold,
                "spike_count": neuron.spike_count,
                "importance": neuron.importance,
                "weights": neuron.weights
            }
        
        # Export anonymous student learning (privacy preserving)
        for student, profile in self.student_profiles.items():
            # Anonymize - just export patterns, not names
            export_data["student_profiles"][f"profile_{hash(student)}"] = {
                "concepts": profile["concepts_strengthened"]
            }
        
        return export_data
    
    def import_knowledge(self, data: Dict):
        """
        Import knowledge from mesh network peers
        """
        # Merge neuron knowledge (weighted averaging)
        if "neurons" in data:
            for neuron_id, neuron_data in data["neurons"].items():
                if neuron_id in self.neurons:
                    # Average the learning
                    current = self.neurons[neuron_id]
                    current.threshold = (current.threshold + neuron_data["threshold"]) / 2
                    
                    for conn, weight in neuron_data["weights"].items():
                        if conn in current.weights:
                            current.weights[conn] = (current.weights[conn] + weight) / 2
        
        # Merge student learning patterns (anonymized)
        if "student_profiles" in data:
            # This is how the network learns collectively!
            pass
    
    # ============================================================
    #  HOMEWORK CHECKING
    # ============================================================
    
    def check_homework(self, question: str, student_answer: str) -> Dict:
        """
        Check homework using spiking pattern matching
        """
        # Generate ideal answer pattern by asking the question
        ideal_response = self.process_spikes(
            self.question_to_spikes(question),
            student_id="teacher"
        )
        
        # Process student answer
        student_response = self.process_spikes(
            self.text_to_spikes(student_answer),
            student_id=self.current_student
        )
        
        # Compare spike patterns
        ideal_spikes = set([a["concept"] for a in ideal_response.get("all_answers", [])])
        student_spikes = set([a["concept"] for a in student_response.get("all_answers", [])])
        
        # Calculate overlap (like Jaccard similarity)
        if not ideal_spikes:
            return {
                "correct": False,
                "score": 0,
                "feedback": "I couldn't understand the question."
            }
        
        overlap = len(ideal_spikes & student_spikes)
        total = len(ideal_spikes | student_spikes)
        
        if total > 0:
            similarity = overlap / total
        else:
            similarity = 0
        
        # Determine if correct
        score = similarity * 100
        correct = score > 60
        
        # Generate feedback
        if correct:
            feedback = "✅ Correct! Great job!"
            if score > 90:
                feedback += " Excellent understanding!"
            else:
                feedback += " Good, but review the details."
        else:
            missing = ideal_spikes - student_spikes
            feedback = f"❌ Not quite. You missed these concepts: {', '.join(list(missing)[:3])}"
            
            # Record learning attempt
            for concept in missing:
                self.record_learning(concept, False)
        
        return {
            "correct": correct,
            "score": score,
            "feedback": feedback,
            "score_out_of_100": int(score),
            "concepts_matched": list(ideal_spikes & student_spikes),
            "concepts_missing": list(ideal_spikes - student_spikes)
        }
